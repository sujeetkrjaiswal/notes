class SnakeNode {
  next = null;
  prev = null;
  x = null;
  y = null;
  constructor(x, y) {
    this.x = x;
    this.y = y;
  }
  toString() {
    return `${this.x},${this.y}`;
  }
}

class SnakeChain {
  _head = null;
  _tail = null;
  _allNodes = new Set();

  addNode(x, y) {
    const node = new SnakeNode(x, y);
    if (this._allNodes.has(node.toString())) {
      throw new Error("ALREADY_EXISTS");
    }

    if (this._head) {
      node.next = this._head;
      this._head.prev = node;
    } else {
      this._tail = node;
    }
    this._head = node;
    this._allNodes.add(node.toString());
  }
  removeLastNode() {
    if (!this._tail) {
      return;
    }
    const lastNode = this._tail
    const newLastNode = lastNode.prev
    this._allNodes.delete(lastNode.toString());
    newLastNode.next = null
    lastNode.prev = null
    lastNode.next = null
    this._tail = newLastNode
  }
  willCollideWithSelf(x, y) {
    const node = new SnakeNode(x, y);
    return this._allNodes.has(node.toString());
  }
  get head() {
    if (!this._head) {
      return { x: null, y: null };
    }
    return { x: this._head.x, y: this._head.y };
  }
  get size() {
    return this._allNodes.size;
  }
  *[Symbol.iterator]() {
    let node = this._head;
    while (node) {
      yield { x: node.x, y: node.y };
      node = node.next;
    }
  }
}

export default class SnakeGame {
  static DIR = {
    LEFT: "LEFT",
    RIGHT: "RIGHT",
    UP: "UP",
    DOWN: "DOWN"
  };
  _state = {
    gameOver: false,
    snakeChain: new SnakeChain(),
    food: {
      x: 0,
      y: 0
    },
    score: 0
  };
  _constraints = {
    vx: 0,
    vy: 0,
    disabledDir: SnakeGame.DIR.LEFT,
    x: 0,
    y: 0
  };
  constructor(tileXCount, tileYCount, initialDir = SnakeGame.DIR.RIGHT) {
    this._constraints.x = tileXCount;
    this._constraints.y = tileYCount;
    this.setDirection(initialDir);
    // initial position of snake
    const x = Math.floor(tileXCount / 2)
    const y = Math.floor(tileYCount / 2)
    this._state.snakeChain.addNode(x,y);
    this._updateFood();
  }
  move() {
    if (this._state.gameOver) return;
    const { x, y } = this._state.snakeChain.head;
    const newX = x + this._constraints.vx;
    const newY = y + this._constraints.vy;

    if (this._willCollide(newX, newY)) {
      this._state.gameOver = true;
      throw new Error('COLLISION')
    }
    this._state.snakeChain.addNode(newX, newY);
    // check if it will eat food
    if (newX === this._state.food.x && newY === this._state.food.y) {
      this._updateFood();
      this._state.score += 1;
    } else {
      this._state.snakeChain.removeLastNode();
    }
  }

  setDirection(dir) {
    if (this._state.gameOver || this._constraints.disabledDir === dir) {
      return false;
    }
    this._constraints.vx = 0;
    this._constraints.vy = 0;
    switch (dir) {
      case SnakeGame.DIR.LEFT:
        this._constraints.vx = -1;
        this._constraints.disabledDir = SnakeGame.DIR.RIGHT
        break;
      case SnakeGame.DIR.RIGHT:
        this._constraints.vx = 1;
        this._constraints.disabledDir = SnakeGame.DIR.LEFT
        break;
      case SnakeGame.DIR.UP:
        this._constraints.vy = -1;
        this._constraints.disabledDir = SnakeGame.DIR.DOWN
        break;
      case SnakeGame.DIR.DOWN:
        this._constraints.vy = 1;
        this._constraints.disabledDir = SnakeGame.DIR.UP
        break;
    }
    return true
  }
  get snakeChain() {
    return [...this._state.snakeChain]
  }
  get score() {
    return this._state.score
  }
  get isGameOver() {
    return this._state.gameOver
  }
  get snakeSize() {
    return this._state.snakeChain.size
  }
  get food() {
    return {x: this._state.food.x, y: this._state.food.y}
  }

  _updateFood() {
    if (this._state.gameOver) return;
    while (true) {
      const x = Math.floor(Math.random() * this._constraints.x);
      const y = Math.floor(Math.random() * this._constraints.y);
      if (!this._state.snakeChain.willCollideWithSelf(x, y)) {
        this._state.food.x = x;
        this._state.food.y = y;
        break;
      }
    }
  }
  _willCollide(x, y) {
    return (
      this._state.snakeChain.willCollideWithSelf(x, y) ||
      x < 0 ||
      x >= this._constraints.x ||
      y < 0 ||
      y >= this._constraints.y
    );
  }
}

### index.js

```javascript
import SnakeGame from "./snake.js";

let gameInstance = null;
const statsPoints = document.querySelector("#stats-points");
const statsDirection = document.querySelector("#stats-direction");
const arena = document.querySelector("#arena");
const gameCtx = arena.getContext("2d");
const config = {
  tileX: 20,
  tileY: 20,
  width: arena.width,
  height: arena.height,
  sizeX: arena.width / 20,
  sizeY: arena.width / 20,
  autoTimer: 1000 / 16
};
let timer;
function init() {
  // update the config object if you accept tile count from user
  gameInstance = new SnakeGame(config.tileX, config.tileY);
  render();
}
function render() {
  gameCtx.fillStyle = "#666";
  gameCtx.fillRect(0, 0, config.width, config.height);
  const food = gameInstance.food;
  gameCtx.fillStyle = "blue";
  gameCtx.fillRect(
    food.x * config.sizeX,
    food.y * config.sizeY,
    config.sizeX,
    config.sizeY
  );
  const state = gameInstance.snakeChain;
  state.forEach(({ x, y }, i) => {
    gameCtx.moveTo(x * config.sizeX, y * config.sizeY);
    gameCtx.fillStyle = i ? "green" : "red";
    gameCtx.fillRect(
      x * config.sizeX,
      y * config.sizeY,
      config.sizeX,
      config.sizeY
    );
  });
}

function autoMove(enabled = true) {
  timer = setTimeout(() => {
    if (gameInstance) {
      try{
        gameInstance.move();
      } catch (err) {
        console.error(err)
      } finally {
        render();
      }
      if (enabled) {
        autoMove(enabled);
      }
    }
  }, config.autoTimer);
}

function debugState() {
  console.dir(gameInstance._state.snakeChain);
  console.dir(gameInstance.food);
}
init();
// Event listeners for keydown
document.addEventListener("keydown", e => {
  switch (e.keyCode) {
    case 32: // space bar
      autoMove(false);
      debugState();
      break;
    case 37: // left
      if(gameInstance.setDirection(SnakeGame.DIR.LEFT)) {
        autoMove(false);
      }
      break;
    case 38: // up
    if(gameInstance.setDirection(SnakeGame.DIR.UP)) {
      autoMove(false);
    }
      break;
    case 39: // right
    if(gameInstance.setDirection(SnakeGame.DIR.RIGHT)) {
      autoMove(false);
    }
      break;
    case 40: // down
    if(gameInstance.setDirection(SnakeGame.DIR.DOWN)) {
      autoMove(false);
    }
      break;
  }
});

// dom event listeners
document.querySelector("#btn-restart").addEventListener("click", e => {
  init();
});
document.querySelector("#btn-move").addEventListener("click", e => {
  autoMove(false);
});
document.querySelector("#btn-debug").addEventListener("click", e => {
  debugState();
});
document.querySelector("#btn-render").addEventListener("click", e => {
  render();
});
```
