# Queue

## Implementation using Single LinkList

```javascript
class LinkedListQueueNode {
    constructor(data, id, next = null) {
        this.data = data
        this.id = id
        this.next = next
    }
    toString() {
        return `(id:${this.id}, data:${this.data})`
    }
}

class LinkedListQueue {
    constructor() {
        this.enqueueRef = null;
        this.dequeueRef = null;
        this.enqueueCounter = -1;
    }

    enqueue(data) {
        const node = new LinkedListQueueNode(data, ++this.enqueueCounter, null)
        if(this.enqueueRef !== null) {
            this.enqueueRef.next = node;
        }
        this.enqueueRef = node
        // if dequeue Ref == null, which only happens when there is nothing to process.
        if(this.dequeueRef === null) {
            this.dequeueRef = this.enqueueRef
        }
    }
    dequeue() {
        const node = this.dequeueRef
        if(this.enqueueRef === node) {
            // if the enqueueRef is pointing to last element, set it to null
            this.enqueueRef = null
            this.dequeueRef = null
        }

        if(node && node.next) {
            this.dequeueRef = node.next
        }
        return node && node.data // no need to return internal structure of the program. Only what you asked from the user as input
    }
    debug(data) {
        // this method is just for debugging. it takes optional parameter over which it decides to either enqueue or dequeue.
        if(data !== undefined) {
            this.enqueue(data)
            console.log(`ACTION ENQUEUE: ${this.enqueueRef.toString()}`)
        } else {
            console.log(`ACTION DEQUEUE: ${this.dequeueRef && this.dequeueRef.toString()}`)
            const nodeData = this.dequeue()
        }
        console.log(`
        QUEUE: ${this.toString()}
        DEQUEUE: ${this.dequeueRef && this.dequeueRef.toString()}
        ENQUEUE: ${this.enqueueRef && this.enqueueRef.toString()}
        `)
    }
    get isEmpty() {
        return this.enqueueRef === null
    }

    get inQueueCount() {
        if(this.enqueueRef === null || this.dequeueRef === null) {
            return 0
        }
        return this.enqueueRef.id - this.dequeueRef.id
    }
    *[Symbol.iterator]() {
        let node = this.dequeueRef
        while(node) {
            yield node.data
            node = node.next
        }
    }
    toString() {
        return [...this].join(' -> ')
    }
}
```
