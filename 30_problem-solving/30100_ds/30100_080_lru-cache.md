# Lru Cache

```javascript
/**
 * LRU: Least Recently Used
 * In the linkedList, the most recent is in the head, the least recent on the end
 */

 class LRUCacheNode {
     constructor(key, data, prev = null, next = null) {
         this.key = key
         this.data = data
         this.next = next
         this.prev = prev
     }
     toString() {
         return `(${this.key}:${this.data})`
     }
 }

 class LRUCache {
    constructor(capacity = 1) {
        this.capacity = capacity
        this.headRef = null // pointer to first node (Most recent)
        this.tailRef = null // pointer to last node (Least recent)
        this.dataMap = new Map()
    }
    deleteNode(node) {
        if(this.headRef === node) {
            this.headRef = node.next
        } else {
            node.prev.next = node.next
        }

        if(this.tailRef === node) {
            this.tailRef = node.prev
        } else {
            node.next.prev = node.prev
        }
    }
    addToHead(node) {
        if(node === this.headRef) {
            return; // Already a head element
        }
        let debugFlag = false
        // if the node is the last element, update tailref to prev element
        if(node === this.tailRef) {
            debugFlag = true
            this.tailRef = node.prev
            this.tailRef.next = null
        }
        node.next = this.headRef
        node.prev = null
        this.headRef.prev = node
        this.headRef = node
    }
    set(key, value) {
        if(this.dataMap.has(key)) { // optional
            // throw new Error('key already exists') // throw error
            this.deleteNode(this.dataMap.get(key)) // or override
        }
        const node = new LRUCacheNode(key, value) // or get the existing node
        this.dataMap.set(key, node)
        if(this.headRef === null) {
            this.headRef = node
        }
        if(this.tailRef === null) {
            this.tailRef = node
        }
        this.addToHead(node)
        if(this.dataMap.size > this.capacity) {
            // remove the least recent used
            this.dataMap.delete(this.tailRef.key)
            this.deleteNode(this.tailRef)

        }
    }
    get(key) {
        if(this.dataMap.has(key)) {
            const node = this.dataMap.get(key)
            this.addToHead(node)
            return {key: node.key, value: node.data}
        }
        console.log(`CACHE MISS: ${key}`)
        return {key, value: null}
    }
    *[Symbol.iterator]() {
        let node = this.headRef
        while(node) {
            yield node
            node = node.next
        }
    }
    toString() {
        return [...this].map(u => u.toString()).join('->')
    }
    debug(key, value) {
        if(value === undefined) {
            console.log(`ACTION: GET (${key})`)
            this.get(key)
        } else {
            console.log(`ACTION: SET (${key}:${value})`)
            this.set(key, value)
        }
        console.log(`
        LRU: ${this.toString()}
        HeadRef: ${this.headRef.toString()}
        TailRef: ${this.tailRef.toString()}
        CachedKeys: ${[...this.dataMap.keys()].join(',')}
        CachedSize: ${this.dataMap.size} used of  ${this.capacity} available
        `)
    }
 }

 test.debug('a',1)
 test.debug('b',1)
 test.debug('c',1)
 test.debug('d',1)
 test.debug('e',1)
 test.debug('f',1)
 test.debug('a')
 test.debug('d')
```
