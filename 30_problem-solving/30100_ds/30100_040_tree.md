## 1. BFS Traversal

class Node {
  next = null;
  constructor(value) {
    this.value = value;
  }
}

class Queue {
  constructor() {
    this.head = null;
    this.tail = null;
  }
  enqueue(value) {
    const node = new Node(value);
    if (this.tail === null) {
      this.head = node;
      this.tail = node;
    } else {
      this.tail.next = node;
      this.tail = node;
    }
  }

  dequeue() {
    if (this.head === null) {
      throw new Error("Nothing to dequeue");
    }
    const node = this.head;
    this.head = node.next;
    node.next = null; // optional
    if (this.head === null) {
      this.tail = null;
    }
    return node.value;
  }
  isEmpty() {
    return this.head === null;
  }
  *[Symbol.iterator]() {
    let node = this.head;
    while (node) {
      yield node.value;
      node = node.next;
    }
  }
}

class Stack {
  head = null;
  tail = null;
  push(value) {
    const node = new Node(value);
    node.next = this.head;
    this.head = node;
    if (this.tail === null) {
      this.tail = this.head;
    }
  }
  pop() {
    if (this.head === null) {
      throw new Error("Nothing to dequeue");
    }
    const node = this.head;
    this.head = node.next;
    node.next = null; // optional
    if (this.head === null) {
      this.tail = null;
    }
    return node.value;
  }
  isEmpty() {
    return this.head === null;
  }
}

class TreeNode {
  constructor(value, left = null, right = null) {
    this.value = value;
    this.left = left;
    this.right = right;
  }
}
class Tree {
  constructor(node) {
    this.root = node;
  }
  bfs() {
    const queue = new Queue();
    const values = [];
    queue.enqueue(this.root);
    while (!queue.isEmpty()) {
      const node = queue.dequeue();
      values.push(node.value);
      if (node.left) {
        queue.enqueue(node.left);
      }
      if (node.right) {
        queue.enqueue(node.right);
      }
    }
    return values;
  }
  dfsPre(values = [], node = this.root) {
    values.push(node.value);
    if (node.left) {
      this.dfsPre(values, node.left);
    }
    if (node.right) {
      this.dfsPre(values, node.right);
    }
    return values;
  }
  dfsIn(values = [], node = this.root) {
    if (node.left) {
      this.dfsPre(values, node.left);
    }
    values.push(node.value);
    if (node.right) {
      this.dfsPre(values, node.right);
    }
    return values;
  }
  dfsPost(values = [], node = this.root) {
    if (node.left) {
      this.dfsPre(values, node.left);
    }
    if (node.right) {
      this.dfsPre(values, node.right);
    }
    values.push(node.value);
    return values;
  }
  dfsPreIter() {
    const stack = new Stack();
    const values = [];
    stack.push(this.root);
    while (!stack.isEmpty()) {
      const node = stack.pop();
      values.push(node.value);
      if (node.right) {
        stack.push(node.right);
      }
      if (node.left) {
        stack.push(node.left);
      }
    }
    return values;
  }
  dfsInIter() {
    const stack = new Stack()
    const values = []
    let node = this.root
    while(true) {
      if(node !== null) {
        stack.push(node)
        node = node.left || null
      } else if(!stack.isEmpty()){
        node = stack.pop()
        values.push(node.value)
        node = node.right
      } else {
        break
      }
    }
    return values
  }
